# Sidebar

### Basic usage

<Playground>
  <Component initialState={{ sidebarVisible: false }}>
    {({ setState, state }) => (
      <React.Fragment>
        <Button onClick={() => setState(state => ({ sidebarVisible: !state.sidebarVisible }))}>Toggle sidebar</Button>
        <Sidebar as={Portal} visible={state.sidebarVisible}>
          <H1>Hello!</H1>
        </Sidebar>
        <Backdrop
          onClick={() => setState(state => ({ sidebarVisible: !state.sidebarVisible }))}
          as={Portal}
          visible={state.sidebarVisible}
        />
      </React.Fragment>
    )}
  </Component>
</Playground>
