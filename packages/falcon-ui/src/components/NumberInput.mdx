# NumberInput

### Basic usage

<Playground>
  <FlexLayout mb="lg" flexDirection="column" alignItems="flex-start">
    Uncontrolled:
    <NumberInput defaultValue={4} mb="md" />

      {({ setState, state }) => (
        <React.Fragment>
          <NumberInput
            value={state.value}
            onChange={e => {
              setState({ value: e.target.value });
            }}
          />
        </React.Fragment>
      )}
    </Component>
  </FlexLayout>
